26    void ins(int x){
27        for(int i=1;i<=len+1;i++){
28            int pos=V[i].size();
29            while(pos>0 && V[i][pos-1]>x){
30                if(pos<V[i].size()) REC.push_back({x,V[i][pos],V[i][pos-1],0});
31                pos--;
32            }
33            if(pos==V[i].size()){
34                V[i].push_back(x);
35                len=max(len,i);
36                break;
37            }
38            if(pos) REC.push_back({V[i][pos-1],V[i][pos],x,1});
39            swap(x,V[i][pos]);
40        }

126    int c[N];
127    int lowbit(int x){
128        return x & (-x);
129    }
130    void modify(int x,int k,int n){
131        while(x<=n) c[x]=max(c[x],k),x+=lowbit(x);
132    }
133    int query(int x){
134        int tot=0;
135        while(x) tot=max(tot,c[x]),x-=lowbit(x);
136        return tot;
137    }

140    int solve(int *a,int n,int lim){
141        for(int i=1;i<=lim;i++) c[i]=0;
142        int tot=0;
143        for(int i=1;i<=n;i++){
144            dp[i]=query(a[i])+1;
145            modify(a[i],dp[i],lim);
146            tot=max(tot,dp[i]);
147        }
148        return tot;
