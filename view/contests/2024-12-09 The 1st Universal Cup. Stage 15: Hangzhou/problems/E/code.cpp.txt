void add(int u,int v,int f){
    e[++cnt]={v,head[u],f};
    head[u]=cnt;

void qadd(int u,int v,int f){
    add(u,v,f);
    add(v,u,0);

int siz,st,ed;
int dis[N];
queue <int> q;
bool bfs(){
    for(int i=1;i<=siz;i++) dis[i]=INF;
    dis[st]=0;
    q.push(st);
    while(q.size()){
        int u=q.front(); q.pop();
        for(int i=head[u];i;i=e[i].nxt){
            int v=e[i].to;
            if(!e[i].f || dis[v]<INF) continue;
            dis[v]=dis[u]+1;
            q.push(v);
        }
    }
    return dis[ed]<INF;
}

int cur[N];
int dfs(int u,int flow){
    if(u==ed) return flow;
    int del=flow;
    for(int i=cur[u];i && del;i=e[i].nxt){
        cur[u]=i;
        int v=e[i].to;
        if(e[i].f && dis[v]==dis[u]+1){
            int x=dfs(v,min(del,e[i].f));
            del-=x;
            e[i].f-=x,e[i^1].f+=x;
            
        }
        
    }
    return flow-del;
}

int dinic(){
    int tot=0;
    while(bfs()){
        for(int i=1;i<=siz;i++) cur[i]=head[i];
        tot+=dfs(st,INF);
    }
    return tot;
}
