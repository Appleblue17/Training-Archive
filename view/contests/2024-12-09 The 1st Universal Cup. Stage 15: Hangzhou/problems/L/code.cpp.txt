using namespace std;
typedef long long LL;
const int N = 1e5 + 10;
int n, q;
LL a[N], gd[17][N];
LL P[70], S[70];
LL pre[70][N], suf[70][N];
int ld[N], tl, rd[N], tr;
void build_st() {
	for(int i = 1; i <= n; i ++ ) gd[0][i] = a[i];
	for(int i = 1; (1 << i) <= n; i ++ ) 
		for(int j = 1; j + (1 << i) - 1 <= n; j ++ ) 
		    gd[i][j] = __gcd(gd[i - 1][j], gd[i - 1][j + (1 << i - 1)]);
}
LL query(int l, int r) {
	if(l > r) return 0;
	int k = log2(r - l + 1);
	return __gcd(gd[k][l], gd[k][r - (1 << k) + 1]);
}
LL solve(int l, int r, int c) { // 首先找到 [l, r] 所有断点 
    tl = 0; ld[++ tl] = l;
    int p = l;
	LL res = 0;
    while(p <= r) {
    	int ll = p, rr = n, mid, res = n + 1;
    	while(ll <= rr) {
    		mid = (ll + rr >> 1);
    		if(query(l, p) != query(l, mid)) res = mid, rr = mid - 1;
    		else ll = mid + 1;
		}
		if(res <= r) ld[++ tl] = res;
		p = res;
	}
    tr = 0; rd[++ tr] = r;
    p = r, res = 0;
    while(p >= l) {
    	int ll = 1, rr = p, mid, res = 0;
    	while(ll <= rr) {
    		mid = (ll + rr >> 1);
    		if(query(p, r) != query(mid, r)) res = mid, ll = mid + 1;
    		else rr = mid - 1;
		}
		if(res >= l) rd[++ tr] = res;
		p = res;
	}
	
	int now = l - 1; LL tmp = 0;
	for(int i = 1; i <= tl; i ++ ) {
		while(now < ld[i] - 1) {
			now ++;
			tmp = __gcd(tmp, a[now]);
		}
		P[i] = tmp;
	}
	now = r + 1; tmp = 0;
	for(int i = 1; i <= tr; i ++ ) {
		while(now > rd[i] + 1) {
			now --;
			tmp = __gcd(tmp, a[now]);
		}
		S[i] = tmp;
	}

		}

	}
		for(int i = 1; i <= tl; i ++ ) {
			for(int j = 1; j <= tr; j ++ ) {
				if(ld[i] >= rd[j]) break;
				res = max(res, __gcd(__gcd(P[i], pre[i][rd[j] - 1]), S[j]));
			}
		}
	}
	else { // 删三个 
for(int i = 1; i <= tl; i ++ ) {
		pre[i][ld[i]] = 0;
		for(int j = ld[i] + 1; j <= rd[1]; j ++ ) pre[i][j] = __gcd(pre[i][j - 1], a[j]);
	}
	for(int i = 1; i <= tr; i ++ ) {
		suf[i][rd[i]] = 0;
		for(int j = rd[i] - 1; j >= ld[1]; j -- ) suf[i][j] = __gcd(suf[i][j + 1], a[j]);
	}
		for(int i = 1; i <= tl; i ++ ) {
			for(int j = 1; j <= tr; j ++ ) {
				if(ld[i] >= rd[j] - 1) break;
				LL tmp = __gcd(P[i], S[j]);

					res = max(res, __gcd(tmp, __gcd(pre[i][k - 1], suf[j][k + 1])));
				}
			}
		}
	}
	return res;
}
