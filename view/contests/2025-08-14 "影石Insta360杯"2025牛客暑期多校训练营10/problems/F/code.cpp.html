12void upd(int o,int x){
13    tag[o]+=x;
14    f[o]+=x;
15}
16void pushdown(int o){
17    int ls=o<<1,rs=o<<1|1;
18    upd(ls,tag[o]);
19    upd(rs,tag[o]);
20    tag[o]=0;
21}
22void pushup(int o){
23    int ls=o<<1,rs=o<<1|1;
24    if(f[ls]>f[rs]) f[o]=f[ls],g[o]=g[ls];
25    else f[o]=f[rs],g[o]=g[rs];
26}
27void build(int l,int r,int o){
28    f[o]=g[o]=tag[o]=0;
29    if(l==r){
30        f[o]=0;
31        g[o]=l;
32        return ;
33    }
34    int mid=(l+r)>>1;
35    build(l,mid,o<<1);
36    build(mid+1,r,o<<1|1);
37    pushup(o);
38}
39void modify(int l,int r,int o,int L,int R,int x){
40    if(L<=l && r<=R){
41        upd(o,x);
42        return ;
43    }
44    pushdown(o);
45    int mid=(l+r)>>1;
46    if(L<=mid) modify(l,mid,o<<1,L,R,x);
47    if(R>mid) modify(mid+1,r,o<<1|1,L,R,x);
48    pushup(o);
49}
