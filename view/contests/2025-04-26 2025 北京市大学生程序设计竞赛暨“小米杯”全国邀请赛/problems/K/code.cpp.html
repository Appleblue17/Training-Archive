40    void ntt(int *f,int n,int opt){
41        for(int i=0;i<n;i++)
42            if(i<rev[i]) swap(f[i],f[rev[i]]);
43        for(int l=1;l<n;l*=2){
44            for(int i=0;i<n;i+=l*2){
45                int u=n/l/2,nw=(opt==1)?0:n;
46                for(int j=0;j<l;j++){
47                    int x=f[i+j],y=pre[nw]*f[i+j+l]%mod;
48                    gmod(f[i+j]=x+y);
49                    gmod(f[i+j+l]=x+mod-y);
50                    nw+=opt*u;
51                }
52            }
53        }
54        if(opt==-1){
