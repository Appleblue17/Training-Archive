8int ksm(int a,int x){
9    int tot=1;
10    while(x){
11        if(x & 1) tot=1ll*tot*a%mod;
12        a=1ll*a*a%mod;
13        x>>=1;
14    }
15    return tot;
16}
17int mul[N],inv[N];
18void init(int lim){
19    mul[0]=inv[0]=1;
20    for(int i=1;i<=lim;i++) mul[i]=1ll*mul[i-1]*i%mod;
21    inv[lim]=ksm(mul[lim],mod-2);
22    for(int i=lim-1;i>=1;i--) inv[i]=1ll*inv[i+1]*(i+1)%mod;
23}
