struct node{
    int x,id;
    int f=0;//1=l,-1=r
} a[N];

ll ksm(ll a,ll b,ll mo){
    ll ret=1,base=a;
    while(b>0){
        if(b&1)ret = ret*base%mo;
        base = base*base%mo;
        b/=2;
    }
    return ret%mo;
}


ll n;

inline void read(ll &x){
    x=0; char c=getchar();
    bool flag=0;
    if(c=='-') flag=1;
    while(c<'0'||c>'9') {
        c=getchar();
        if(c=='-') flag=1;
    }
    while(c>='0'&&c<='9') x=(x<<1)+(x<<3)+(c-'0'), c=getchar();
    if(flag) x=-x;
}

bool cmp(node a,node b){
    if(a.x != b.x)return a.x < b.x;
    else return a.f > b.f;
}
