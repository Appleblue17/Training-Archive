                    for(int i=0;i<=j;i++){
                        for(int k=0;k<=n;k++){
                            dp[t][i][j][k]=-1;
                        }
                    }
                }
            }
            
            dp[0][0][0][0]=0;
            for(int t=0;t<=lim;t++){
                int plim=(t<=n)?n:0;
                for(int j=0;j<=c;j++){
                    for(int i=0;i<=j;i++){
                        for(int k=0;k<=n;k++){
                            if(dp[t][i][j][k]==-1) continue;

                            }

                }
