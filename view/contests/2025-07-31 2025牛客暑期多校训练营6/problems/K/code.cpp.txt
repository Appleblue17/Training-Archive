            for(int j=x;j<=lim;j+=x) D[j].push_back(x);
            if(1ll*x*i>lim) break;
        }
    }
}
int cal(int x,int y){
    return __gcd(abs(x),abs(y));
}
 
 
int T,n;

int f[19][N];
int qry(int l,int r){
    if(l>r) return 0;
    int x=log2(r-l+1);
    return cal(f[x][l],f[x][r-(1<<x)+1]);
}
