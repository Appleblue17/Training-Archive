85struct DSU{
86	int fa[N], sz[N], sum[N];
87	inline int get(int x){
88		if(x==fa[x]) return x;
89		return get(fa[x]);
90	}
91	struct node{
92		int rfx, rfy;
93		int rszx, rszy;
94		int rsumx, rsumy;
95	};
96	vector<node> stk;
97	void merge(int x, int y){
98		x=get(x); y=get(y);
99		if(x==y) return ;
100		if(sz[x]<sz[y]) swap(x, y);
101		stk.push_back((node){x, y, sz[x], sz[y], sum[x], sum[y]});
102		sz[x]+=sz[y]; sum[x]+=sum[y]; fa[y]=x;
103	}
104	void back(int x){
105		while((int)stk.size()>x){
106			node cur=stk.back(); stk.pop_back();
107			fa[cur.rfx]=cur.rfx; fa[cur.rfy]=cur.rfy;
108			sz[cur.rfx]=cur.rszx; sz[cur.rfy]=cur.rszy;
109			sum[cur.rfx]=cur.rsumx; sum[cur.rfy]=cur.rsumy;
110		}
111	}
112	void init(){
113		for(int i=1; i<=n; ++i) fa[rid(i, 0)]=rid(i, 0), sz[rid(i, 0)]=1, sum[rid(i, 0)]=1;

115		for(int i=1; i<=n2; ++i) fa[rid(i, 2)]=rid(i, 2), sz[rid(i, 2)]=1, sum[rid(i, 2)]=0;
