struct info{

    int pre0, suf0, siz, st, ed;
    void set(int o){

            all0=1;
            pre0=1; suf0=1; siz=0; st=0; ed=0;

        else{
            all0=0;

    info z=empty_info;
    z.all0=x.all0&y.all0;

void mdf(int p, int l, int r, int x){

}
