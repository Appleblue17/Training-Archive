32void sol(int t,int mac,int l,int r,int L,int R){

34    ans=max(ans,mac);

37    int A=l,B=r;
38    while(A<=B){
39        while(A<=B && (p[A]>>t & 1ll)) A++;

41        if(A<=B) swap(p[A],p[B]);

46    while(A<=B){

56
