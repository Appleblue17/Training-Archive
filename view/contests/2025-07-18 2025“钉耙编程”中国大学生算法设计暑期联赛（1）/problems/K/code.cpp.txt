void add1(int col, int l, int r){
	int sum=0;
	for(int i=1; i<=pos[n]; ++i){
		int cl=max(lp[i], l), cr=min(rp[i], r);
		sum+=max(0, cr-cl+1);
		occ[col][i]+=sum;
	}
}
void del1(int col, int l, int r){
	int sum=0;
	for(int i=1; i<=pos[n]; ++i){

		sum+=max(0, cr-cl+1);

void add2(int col){

		}
	}
