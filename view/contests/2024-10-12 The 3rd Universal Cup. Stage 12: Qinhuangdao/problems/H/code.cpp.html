53int cur[N];
54int dfs(int u,int flow){
55    if(u==ed) return flow;
56    int del=flow;
57    for(int i=cur[u];i && del;i=e[i].nxt){
58        cur[u]=i;

60        if(e[i].f && dis[v]==dis[u]+1){
61            int x=dfs(v,min(del,e[i].f));

65    }
