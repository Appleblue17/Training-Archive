95bool ckvalid(p2 st,p2 ed){
96    for(rect v:rt){
97        if((st == p2(v.x,v.y)) && (ed == p2(v.X,v.Y)))return 0;
98        if((st == p2(v.x,v.Y)) && (ed == p2(v.X,v.y)))return 0;
99        if(segmentPI(st,ed,p2(v.x,v.y),p2(v.x,v.Y)))return 0;
100        if(segmentPI(st,ed,p2(v.x,v.Y),p2(v.X,v.Y)))return 0;
101        if(segmentPI(st,ed,p2(v.X,v.Y),p2(v.X,v.y)))return 0;
102        if(segmentPI(st,ed,p2(v.X,v.y),p2(v.x,v.y)))return 0;
103
104        if(segmentPI(st,ed,p2(v.x,v.y),p2(v.X,v.Y)))return 0;
105        if(segmentPI(st,ed,p2(v.x,v.Y),p2(v.X,v.y)))return 0;
106
107        if(st.x > v.x+eps && st.x < v.X-eps && st.y > v.y+eps && st.y < v.Y-eps)return 0;
108        if(ed.x > v.x+eps && ed.x < v.X-eps && ed.y > v.y+eps && ed.y < v.Y-eps)return 0;
