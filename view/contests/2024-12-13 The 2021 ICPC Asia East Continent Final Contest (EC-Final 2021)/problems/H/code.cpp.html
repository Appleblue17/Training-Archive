145inline void upd(int x, int y, int z){
146    while(top[x]!=top[y]){
147        if(dep[top[x]]<dep[top[y]]) swap(x, y);
148        for(auto t:bigchain[top[x]]) {
149            if(dfn[t]<=dfn[x]) {
150                bigcnt[t]+=z;
151                summx-=bigmx[fa[t]];
152                bigmx[fa[t]]=max(bigmx[fa[t]], bigcnt[t]);
153                summx+=bigmx[fa[t]];
154            }
155        }
156        add(1, 1, n, dfn[top[x]], dfn[x], z);
157        fnd(1, 1, n, dfn[top[x]], dfn[x]);
158        x=fa[top[x]];
159    }
160    if(dfn[x]>dfn[y]) swap(x, y);
161    for(auto t:bigchain[top[x]]){
162        if(dfn[t]>=dfn[x]+1&&dfn[t]<=dfn[y]) {

171}
