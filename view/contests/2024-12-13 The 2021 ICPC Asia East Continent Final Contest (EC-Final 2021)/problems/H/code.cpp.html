50int lin[M], nxt[M], to[M], val[M], tot=1;
51inline void in(int x, int y, int z){
52    nxt[++tot]=lin[x]; lin[x]=tot; to[tot]=y; val[tot]=z;
53    nxt[++tot]=lin[y]; lin[y]=tot; to[tot]=x; val[tot]=0;

66inline bool bfs(){
67    for(int i=1; i<=idx; ++i) dis[i]=inf, cur[i]=lin[i];
68    dis[S]=0; q[hh=tt=1]=S;

80inline int dfs(int x, int flow){
81    if(x==T) return flow;
82    int del=flow;
83    for(int &i=cur[x]; i&&del; i=nxt[i]){
84        int y=to[i];
85        if(val[i]&&dis[y]==dis[x]+1){
86            int z=dfs(y, min(del, val[i]));
87            del-=z;
88            val[i]-=z; val[i^1]+=z;
89        }
90    }
91    return flow-del;
92}
