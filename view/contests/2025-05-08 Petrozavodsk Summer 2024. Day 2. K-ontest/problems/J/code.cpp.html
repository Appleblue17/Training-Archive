32void dijkstra(){
33    for(int i=1;i<=id;i++) dis[i]=INF;
34    dis[1]=0;
35    q.push((def){1,0});
36    
37    while(!q.empty()){
38        def tmp=q.top();
39        q.pop();
40        int u=tmp.num;
41        if(vis[u]) continue;
42        vis[u]=1;
43        for(int i=head[u];i;i=e[i].nxt){
44            int v=e[i].to;
45            if(!vis[v] && dis[u]+e[i].w<dis[v]){
46                dis[v]=dis[u]+e[i].w;
47                q.push((def){v,dis[v]});
48            }
49        }
50    }
51}
