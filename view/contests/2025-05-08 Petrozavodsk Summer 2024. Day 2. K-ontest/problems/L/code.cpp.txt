using namespace std;
#define int long long
const int N=1e6+5,M=330,INF=1e18;
int n,s,r,mx;
int val[M];

struct abc{
    int x,y;
};
bool operator <(abc X,abc Y){
    return (__int128)X.x*(__int128)Y.y<(__int128)Y.x*(__int128)X.y;
}
abc min(abc X,abc Y){
    return (X<Y)?X:Y;
}
abc max(abc X,abc Y){
    return (X<Y)?Y:X;
}
void reduce(abc &X){
    int g=__gcd(X.x,X.y);
    X=(abc){X.x/g,X.y/g};
}

void pr(abc X){
    cout<<X.x<<"/"<<X.y;
}

int dp[N];

signed main() {
    cin>>n>>s>>r;
    for(int i=1;i<=n;i++){
        int A,B;
        scanf("%lld%lld",&A,&B);
        val[A]=max(val[A],B);
        mx=max(mx,A);
    }
    
    for(int i=1;i<N;i++) dp[i]=-INF;
    for(int i=1;i<=mx;i++){
        if(val[i]){
            for(int j=0;j+i<N;j++){
                dp[j+i]=max(dp[j+i],dp[j]+val[i]);
            }
        }
    }
    // for(int i=0;i<=10;i++) cout<<dp[i]<<" ";

    abc mn={1,0};
    int mner;
    for(int i=1;i<=mx;i++){
        if(val[i]){
            abc tmp={i,val[i]};
            if(tmp<mn){
                mn=tmp;
                mner=i;
            }
        }
    }

    int a0=mner,b0=val[mner];

    int lim=ceil(lim0);
    // cout<<"lim: "<<lim0<<" "<<lim<<endl;

    // cout<<K<<endl;
    
    abc ans={1,0};
    int mink,mini;

                ans=tmp;
                mink=k,mini=i;
            }
