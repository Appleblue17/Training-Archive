int calA(int pos){
    return f[pos]+g[pos]-a[pos]*2;
}
int calB(int pos){
    if(S[pos]=='1') return INF;
    int mn=INF,tot=INF;
    
    int L=pos,R=pos;
    while(L-1>=pos-k+1 && S[L]=='0') L--;
    while(R+1<=pos+k-1 && S[R]=='0') R++;
    
    // cout<<pos<<" "<<L<<" "<<R<<endl;
    
    for(int i=pos-k+1;i<=pos-1;i++){
        int l=pos+1,r=i+k;
        if(r<=R) mn=min(mn,g[r]);
        if(i>=L) tot=min(tot,f[i]+mn);
        // cout<<f[i]<<" "<<mn<<endl;
    }
    return tot;
}

        for(int i=1;i<=n+1;i++){

            if(S[i]=='1'){
                head=tail=1;

                que[++tail]=i;
