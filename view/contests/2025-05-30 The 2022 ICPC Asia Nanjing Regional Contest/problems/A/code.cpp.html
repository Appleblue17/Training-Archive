6int sum[N][N];
7inline int get(int u, int d, int l, int r){
8    return sum[d][r]-sum[u-1][r]-sum[d][l-1]+sum[u-1][l-1];
9}

22            --up; --dp;

32        }
33    }

61    }
62    for(int i=0; i<=n*2+1; ++i){
63        for(int j=0; j<=m*2+1; ++j){
64            sum[i][j]+=sum[i][j-1]+sum[i-1][j]-sum[i-1][j-1];
65        }
66    }

69        for(int j=0; j<=m-1; ++j){

71                ++ans;
72            }
73        }
74    }
75    printf("%d\n", ans);
76}
