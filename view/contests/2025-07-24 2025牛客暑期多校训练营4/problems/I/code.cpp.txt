const int dir[4][2] = {{1,0},{0,-1},{0,1},{-1,0}};
  
int a[N][N],tar[N][N],pos[N][N],vis[N][N],dis[N][N];
 
vector<int> fans[3];
 
void addans(int x,int y,int z){
    //assert(pos[x][y]==1);
    //assert(pos[x+dir[z][0]][y+dir[z][1]]==0);
    if(pos[x+dir[z][0]][y+dir[z][1]]!=0)return;
    pos[x][y]=0;
    pos[x+dir[z][0]][y+dir[z][1]]=1;
    fans[0].push_back(x);
    fans[1].push_back(y);
    fans[2].push_back(z);
}
 
vector<int> sav[3];
 
void saveans(int x,int y,int z){
    sav[0].push_back(x);
    sav[1].push_back(y);
    sav[2].push_back(z);
}
  
int n,m,k;
char s[N];
  
void solve(){

    for(int box=1;box<=50;box++){
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            queue<pii> Q;
            if(!pos[i][j])continue;
            if(pos[i][j] && tar[i][j])continue;

            Q.push(mp(i,j));

                    }
                }
            }
            if(ans == 0){
