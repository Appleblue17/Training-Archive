using namespace std;
const int N=2200,Q=1e6+5;
int n,q;
int A[N][N],B[N][N];

int OPT[Q],VAL[Q],ANS[Q];

int rev(int x){
    int tot=0;
    while(x){
        tot=tot*10+x%10;
        x/=10;
    }
    return tot;
}


map<int,int> mp;
int p[N],e[N],id;
void decom(int n){
    for(int i=2;i*i<=n;i++){
        if(n%i) continue;
        while(n%i==0){
            mp[i]++;
            n/=i;
        }
    }
    if(n>1) mp[n]++;
}


vector < pair<int,int> > G[N*2];
void add(int u,int v,int w){
    G[u].push_back({v,w});
    G[v].push_back({u,w});
}
bool FL;
int cid;
int col[N*2],bel[N*2];

void dfs(int u){
    bel[u]=cid;
    for(auto e: G[u]){
        int v=e.first,w=e.second;
        if(col[v]==-1){
            col[v]=col[u]^w;
            dfs(v);
        }
        else{
            if(col[v]!=(col[u]^w)) FL=0;
        }
    }
}

bool nw[N*2];
int buc[N*2][2],ans;

void upd(int pos,int val,int w){
    int c=bel[pos];
    ans-=(buc[c][0]==0)+(buc[c][1]==0);
    if(val==col[pos]) buc[c][1]+=w;
    else buc[c][0]+=w;
    ans+=(buc[c][0]==0)+(buc[c][1]==0);
}
