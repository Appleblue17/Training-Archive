55        void fft(cpl *a,int n,int opt){
56            for(int i=0;i<n;i++)
57                if(i<rev[i]) swap(a[i],a[rev[i]]);
58            for(int l=1;l<n;l*=2){
59                for(int i=0;i<n;i+=l*2){
60                    int u=n/l/2,nw=(opt==1)?0:n;
61                    for(int j=0;j<l;j++){
62                        cpl x=a[i+j],y=pre[nw]*a[i+j+l];
63                        a[i+j]=x+y,a[i+j+l]=x-y;
64                        nw+=opt*u;
65                    }
66                }
67            }
68            if(opt==-1){
69                for(int i=0;i<n;i++) a[i].A=a[i].A/n,a[i].B=a[i].B/n;

73            int M=sqrt(mod);
74            int lim=init(n,m);
75            for(int i=0;i<=n;i++) A[i]=(cpl){f[i]%M,f[i]/M},B[i]=(cpl){f[i]%M,-f[i]/M};
76            for(int i=0;i<=m;i++) C[i]=(cpl){g[i]%M,g[i]/M};
77            for(int i=n+1;i<(1ll<<lim);i++) A[i]=B[i]=(cpl){0,0};
78            for(int i=m+1;i<(1ll<<lim);i++) C[i]=(cpl){0,0};
79            fft(A,(1<<lim),1);
