133vector<Point> ConvexHull(vector<Point> p) {  
134    //  
135    sort(p.begin(), p.end());  
136    p.erase(unique(p.begin(), p.end()), p.end());  
137    int n = p.size(), m = 0;  
138    vector<Point> res(n+1);  
139    for(int i = 0; i < n; i++) {  
140        while(m > 1 && Cross(res[m-1]-res[m-2], p[i]-res[m-2]) <= 0) m--;  
141        res[m++] = p[i];  
142    }  
143    int k = m;  
144    for(int i = n-2; i >= 0; i--) {  
145        while(m > k && Cross(res[m-1]-res[m-2], p[i]-res[m-2]) <= 0) m--;  
146        res[m++] = p[i];  
147    }  
148    m -= n > 1;  
149    res.resize(m);  
150    return res;  
151}
