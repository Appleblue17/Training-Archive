bool operator <(def X,def Y){

    else win[X]--;
    score[X]-=x-y;
    score[Y]-=y-x;
    bbt[col[X]].insert((def){win[X],score[X],rk[X]});
    bbt[col[Y]].insert((def){win[Y],score[Y],rk[Y]});
}

}
