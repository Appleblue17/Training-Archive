84vector<Point> ConvexHull(vector<Point> p) {  
85    //预处理，删除重复点  
86    sort(p.begin(), p.end());  
87    p.erase(unique(p.begin(), p.end()), p.end());  
88    int n = p.size(), m = 0;  
89    vector<Point> res(n+1);  
90    for(int i = 0; i < n; i++) {  
91        while(m > 1 && Cross(res[m-1]-res[m-2], p[i]-res[m-2]) <= 0) m--;  
92        res[m++] = p[i];  
93
94    }  
95    int k = m;  
96    for(int i = n-2; i >= 0; i--) {  
97        while(m > k && Cross(res[m-1]-res[m-2], p[i]-res[m-2]) <= 0) m--;  
98        res[m++] = p[i];  
99
100    }  
101    m -= n > 1;  
102    res.resize(m);  
103    return res;  
104}

112vector<Point> GetCommonTangents(const Circle& A, const Circle& B) {
113    if(A.r < B.r)return GetCommonTangents(B,A);
114    resa.clear();resb.clear();
115    vector<Point> res;//可以改为vector<Line>等，按需要决定
116    Vector v = B.c - A.c;
117    double d = Length(v);
118    if (dcmp(d) == 0) {
119        if (dcmp(A.r - B.r) == 0) {
120            // 同心且半径相等，有无数条公切线，但无法枚举，返回空
121        }
122        // 同心但半径不等，无公切线
123        return res;
124    }
125    double r_sum = A.r + B.r;
126    double r_diff = fabs(A.r - B.r);
127    if (dcmp(d - r_diff) == 0) { // 内切，只有一条外公切线
128        Point p = A.c + v * (A.r / d);
129        Vector dir = Rotate(v, PI / 2);
130        Line line = Line(p, dir);
131        //resa.push_back(p);resb.push_back(p);
132        //res.push_back(p);
133    }
134    // 计算外公切线 (d >= |r1 - r2| 时存在)
135    if (dcmp(d - r_diff) > 0) {//内切已经讨论过了
136        double alpha = acos(r_diff / d);
137        double angles[2] = {alpha,-alpha};
138        //cout << base << ' ' << alpha << endl;
139        for (int i = 0; i < 2; i++) {
140            Vector unit_vec = Rotate(v,angles[i]);

172double PolygonArea(vector<Point> po) {

174    double area = 0.0;
