void bfs(int x){
    queue<int> q;
    q.push(x);
    while(!q.empty()){
        int u = q.front();
        q.pop();
        for(int v:g[u]){
            if(dist[x][v]>=10000){

                dist[x][v]=dist[x][u]+1;

int mton(int m){
    double nd = -0.5+sqrt(8*m*(m+1))/2.0;
    int n = floor(nd);
    return n;
}

double cal0(int n,int x){
    if(!n) return 0;
    int d=x/n,r=x%n;
    return 1.0*(n-r)/(d+1)+1.0*r/(d+2);
}
double cal2(int n,int m,int k){
    if(n>10000 || m>10000) return 1e9;
    if(!n && !m) return 0;
    double x=n*(k+m*(1.0-SQ2))/(SQ2*m+n);
    if(x<0) x=0;
    if(x>k) x=k;
    // cout<<n<<" "<<m<<" "<<k<<": "<<x<<endl;
    
    int t=x;
    double tot=1e18;
    int mn;
    for(int i=-1;i<=2;i++){
        int y=t+i;
        if(y<0 || y>k) continue;
        double val=cal0(n,y)+cal0(m,k-y)*2;
        if(val<tot) tot=val,mn=y;
    }
    // cout<<n<<" "<<m<<" "<<k<<": "<<x<<" "<<mn<<" "<<tot<<endl;
    return tot;
}
